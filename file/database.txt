CREATE TABLE `diary_apply` (
	`seq` INT(10) NOT NULL AUTO_INCREMENT COMMENT '시퀀스 번호',
	`reg_date` TIMESTAMP NULL DEFAULT CURRENT_TIMESTAMP COMMENT '등록 날짜',
	`del_flag` VARCHAR(2) NULL DEFAULT 'N' COMMENT '삭제 여부',
	`user_seq` INT(11) NULL DEFAULT NULL COMMENT '유저 seq',
	`start_date` DATETIME NULL DEFAULT NULL COMMENT '시작 날짜',
	`end_date` DATETIME NULL DEFAULT NULL COMMENT '종료 날짜',
	`title` VARCHAR(255) NULL DEFAULT NULL COMMENT '제목',
	`bg_color` VARCHAR(20) NOT NULL,
	`text_color` VARCHAR(20) NOT NULL,
	PRIMARY KEY (`seq`)
)
COLLATE='utf8_general_ci'
ENGINE=InnoDB
AUTO_INCREMENT=306
;


CREATE TABLE `diary_userInfo` (
	`seq` INT(10) NOT NULL AUTO_INCREMENT COMMENT '시퀀스 번호',
	`reg_date` TIMESTAMP NULL DEFAULT CURRENT_TIMESTAMP COMMENT '등록 날짜',
	`del_flag` VARCHAR(2) NULL DEFAULT 'N' COMMENT '삭제 여부',
	`id` VARCHAR(50) NULL DEFAULT NULL,
	`pw` VARCHAR(100) NULL DEFAULT NULL,
	PRIMARY KEY (`seq`)
)
COLLATE='utf8_general_ci'
ENGINE=InnoDB
AUTO_INCREMENT=7
;
